    private void btnloginActionPerformed(java.awt.event.ActionEvent evt) {                                         
        connection();
        String username = txtusername.getText();
        String password = txtPassword.getText();
        try
        {
        Statement st = cn.createStatement();
        String sql = "select * from tblsecurity where username like '"+username+"' and password like '"+password+"'";
        ResultSet rs = st.executeQuery(sql);
        boolean found = false;
        while (rs.next())
        {
            if(username.equals(rs.getString("username")) && password.equals(rs.getString("password")))
            {
               JOptionPane.showMessageDialog(null, "Login Successfully!");
               found = true;
               new Option().setVisible(true);
               this.dispose();
            }
            if(found = false)
                JOptionPane.showMessageDialog(null, "Invalid account");
        }
        }
        catch(Exception ex)
        {
            JOptionPane.showMessageDialog(null, ex.getMessage());
        }
        
    }

    public void loadrecord(JTable table)
    {
        connection();
        try
        {
            PreparedStatement ps = cn.prepareStatement("Select *from tblfoods");
            ResultSet rs = ps.executeQuery();
            DefaultTableModel model = (DefaultTableModel)table.getModel();
            model.setRowCount(0);
            Object [] row;
            while (rs.next())
            {
                row = new Object[3];
                row [0] = rs.getInt(1);
                row [1] = rs.getString(2);
                row [2] = rs.getInt(3);
                model.addRow(row);
            }
            cn.close();
        }
        catch(Exception ex)
        {
            JOptionPane.showMessageDialog(null, ex.getMessage());
        }
    }

    public void loadrecord(JTable table)
    {
        connection();
        try
        {
            PreparedStatement ps = cn.prepareStatement("Select *from tblfoods");
            ResultSet rs = ps.executeQuery();
            DefaultTableModel model = (DefaultTableModel)table.getModel();
            model.setRowCount(0);
            Object [] row;
            while (rs.next())
            {
                row = new Object[3];
                row [0] = rs.getInt(1);
                row [1] = rs.getString(2);
                row [2] = rs.getInt(3);
                model.addRow(row);
            }
            cn.close();
        }
        catch(Exception ex)
        {
            JOptionPane.showMessageDialog(null, ex.getMessage());
        }
    }

    private void btnSaveActionPerformed(java.awt.event.ActionEvent evt) {                                        
        connection();
        String fud = txtFood.getText();
        String price = txtPrice.getText();
        try
        {
          PreparedStatement ps = cn.prepareStatement("insert into tblfoods (Foods, Price)values (?,?)");  
          ps.setString(1, fud);
          ps.setString(2, price);
          ps.execute();
          loadrecord(jTable1);
          cn.close();
          
          JOptionPane.showMessageDialog(null, "Saved Successfully!");
          txtID.setText("0");
          txtFood.setText("");
          txtPrice.setText("");
        }
        catch(Exception ex)
        {
            JOptionPane.showMessageDialog(null, ex.getMessage());
        }
    }                                       

    private void btnUpdateActionPerformed(java.awt.event.ActionEvent evt) {                                          
        connection();
        try
        {
           String id = txtID.getText();
           String fud = txtFood.getText();
           String price = txtPrice.getText();
           
           PreparedStatement ps = cn.prepareStatement("update tblfoods set Foods = '"+fud+"' , Price = '"+price+"' where id like '"+id+"'");
           ps.execute();
           cn.close();
           JOptionPane.showMessageDialog(null, "Updated successfully!");
           loadrecord(jTable1);
           txtID.setText("0");
           txtFood.setText("");
           txtPrice.setText("");
           
        }
        catch(Exception ex)

            JOptionPane.showMessageDialog(null, ex.getMessage());
        }
    }
